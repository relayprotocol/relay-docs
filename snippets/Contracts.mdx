export const Contracts = ({ env = "mainnet" }) => {

env === "testnets"

setChainData(
response.chains.sort((a, b) =>

})

}, [env]);

      <th>RelayReceiver</th>
      <th>ERC20Router</th>
      <th>ApprovalProxy</th>
    </tr>
  </thead>

    {chainData.map((chain) => {
